/* 掲示板の最新3件を取得して表示するスクリプト */ function loadRecentBBS() { const previewArea = document.getElementById('bbs-preview'); const WORKER_URL = 'https://dry-silence-4f1f.y-bb0.workers.dev'; fetch(WORKER_URL) .then(response => { if (!response.ok) throw new Error('ログが見つかりません'); return response.json(); }) .then(posts => { if (posts.length === 0) { previewArea.innerText = "書き込みはまだありません。"; return; } previewArea.innerHTML = ''; // 最新の3件を表示 for (let i = 0; i < Math.min(posts.length, 3); i++) { const post = posts[i]; // 管理者バッジの判定 const adminIcon = post.isAdmin ? '' : ''; const div = document.createElement('div'); div.className = 'recent-post'; // メッセージ内のHTMLタグをエスケープして安全に表示 const safeName = escapeHtml(post.name); const safeBody = escapeHtml(post.message); div.innerHTML = `${safeName}${adminIcon}${safeBody}`; previewArea.appendChild(div); } }) .catch(err => { previewArea.innerText = "読み込みに失敗しました。"; }); } // HTMLエスケープ用関数 function escapeHtml(str) { if (!str) return ""; return str.replace(/[&<>"']/g, m => ({'&':'&','<':'<','>':'>','"':'"',"'":'''}[m])); } loadRecentBBS();
